	\question Find the least-squares solution of $A\bx=\bb$, where:
	      	      	      	      	          
	      \begin{parts}
	      	\part $A=\mat{2\\3\\5\\7}$, $\bb=\mat{1\\4\\6\\8}$
	      	      	      	      	      	      	      	      	      	      			  
	      	      \begin{solution} We solve the associated normal equations, $A^TA\bx = A^T\bb$:
	      	      \begin{align*}
	      	      	A^TA   & = \mat{2 & 3 & 5 & 7}\mat{2 \\3\\5\\7}=86,\\
	      	      	A^T\bb & = \mat{2 & 3 & 5 & 7}\mat{2 \\4\\6\\8}=100\\
	      	      	\bx &=100/86=50/43
					\end{align*}
				\end{solution}
	      	      	      	      	      	      	      	      
	      	\part $A=\mat{1&0\\1&1\\1&1}$, $\by=\mat{1\\0\\0}$
	      	      	      	      			  
	      	      \begin{solution} We solve the associated normal equations, $A^TA\bx = A^T\bb$:
	      	      \begin{align*}
	      	      	A^TA & = \mat{1 & 1 & 1 \\0&1&1}\mat{1&0\\1&1\\1&1}
	      	      	= \mat{3&2\\2&2},\\
	      	      	A^T\bb & = \mat{1&1&1\\0&1&1}\mat{1\\0\\0}=\mat{1\\0}\\
	      	      	\bx &=\mat{3&2\\2&2}^{-1}\mat{1\\0}\\
	      	      	&=\frac12\mat{2&-2\\-2&3}\mat{1\\0} = \mat{1\\-1}
					\end{align*}
				\end{solution}
	      	      	      	      	      	      
	      \end{parts}
	      	      	      	      	      
	      	      	      	      	      		  
	      	      	      	      	      	      	      	      
	\question Consider the dataset
	      \[
	      	(0, 0),\; (1, 1),\; (2, 3),\; (3, 6).
	      \]
	      \begin{parts}
	      	\part Find the least-squares line for these data, and the associated sum of squared errors (SSE).
	      	      	      	      			  
	      	      \begin{solution}
	      	      Let
	      	      \[
	      	      	X = \mat{1&0\\1&1\\1&2\\1&3},\quad
	      	      	\bbeta = \mat{\beta_0\\\beta_1},\quad
	      	      	\by=\mat{0\\1\\3\\6}.
	      	      \]
	      	      We solve $X^TX\bbeta=X^T\by$.
	      	      \begin{align*}
	      	      	XX^T & = \mat{1 & 1 & 1 & 1 \\0&1&2&3}\mat{1&0\\1&1\\1&2\\1&3} = 
	      	      	\mat{4&6\\6&14}\\
	      	      	X^T\by &= \mat{1 & 1 & 1 & 1 \\0&1&2&3}\mat{0\\1\\3\\6} = \mat{10\\25}\\
	      	      	\bbeta &= (X^TX)^{-1}X^T\bbeta\\
	      	      	&= \frac1{20}\mat{14&-6\\-6&4}\mat{10\\25} = \frac1{20}\mat{-10\\40} = \mat{-1/2\\2}
	      	      \end{align*}
	      	      Thus, the least-squares line associated to these data is
	      	      \[
	      	      	y=-\frac12 + 2x.
	      	      \]
	      	      Since 
	      	      \[
	      	      	y(0)=-\frac12,\quad y(1)=\frac32,\quad y(2)=\frac72,\quad y(3)=\frac{11}2,
	      	      \]
	      	      \[
	      	      	\text{SSE} = \left(-\frac12 - 0\right)^2 + 	\left(\frac32 - 1\right)^2
	      	      	\left(\frac72 - 3\right)^2 + \left(\frac{11}2 - 6\right)^2=1.
					\]
				\end{solution}

	      	\part Find the quadratic equation,
	      	      \[
	      	      	y(x) = \beta_0 + \beta_1x + \beta_2x^2,
	      	      \]
					minimizing the SSE.
					
	      	      \emph{Suggestion:} Find a matrix $X$ such that
	      	      \[
	      	      	X\mat{\beta_0\\\beta_1\\\beta_2}=\mat{y(0)\\y(1)\\y(2)\\y(3)}
	      	      \]
	      	      					
\begin{solution}					Let
					\begin{align*}
						X &= \mat{1&0^1&0^2\\1&1^1&1^2\\1&2^1&2^2\\1&3^1&3^2} =\mat{1&0&0\\1&1&1\\1&2&4\\1&3&9}.
						\intertext{Now solve $X^TX\bbeta=\by$:}
						\bbeta &= \left[\begin{matrix}- \frac{3}{14}\\[1ex]\frac{1}{2}\\[1ex]\frac{5}{7}\end{matrix}\right]
												\end{align*}
				\end{solution}
				\end{parts}

	\question Let
	      \[
	      	A=\mat{1&2&3\\4&5&6\\7&8&9},
	      	\quad \bb=\mat{0\\0\\1}.
	      \]
	      Note that the columns of $A$ are \emph{not} linearly independent.
	      \begin{parts}
	      	\part Find the projection, $\widehat{\bb}$, of $\bb$ onto $C(A)$.
	      	\part Solve the system
	      	      \[
	      	      	A\widehat{\bx}=\widehat{\bb}.
	      	      \]
	      	\part Find the unique solution that belongs to $C(A^T)$.
	      	\part Can you identity the minimal solution, i.e., the one  with smallest length?
	      \end{parts}
